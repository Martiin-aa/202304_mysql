-- Consulta: recupera todos los usuarios que marcaron como favorito el tercer libro
SELECT 
	usuarios.nombre, libros.titulo 
FROM
    libros
        JOIN
    favoritos ON libros.id = favoritos.libro_id
        JOIN
    usuarios ON usuarios.id = favoritos.usuario_id
WHERE
    libro.id = 3;
